# Step 1
    [Option 1]
    Check conda virtual environment
    >>> conda env list
    Activate the virtual environment, e.g.,
    >>> conda activate qml
    One can use `python -V` to double check venv version

    [Option 2]
    See https://stackoverflow.com/questions/61915607/commandnotfounderror-your-shell-has-not-been-properly-configured-to-use-conda
    Add "source ~/anaconda3/etc/profile.d/conda.sh" in shell script
    Then you can call "conda activate your_env_name" in shell script

# Step 3
Write a shell script, e.g., consider `a.sh`
'''
    #!/bin/bash
    #SBATCH --exclusive
    python_version="$(python --version)"
    required_version="Python 3.9.12"
    echo current python version is $python_version
    echo required python version is $required_version
    if [ "$python_version" = "$required_version" ]; then
        echo "Python 3.9.12 detected"
        python a.py
    else
        echo "Python 3.9.12 not detected"
    fi
'''

# Step 4
Run `a.sh` with sbatch
>>> sbatch -p large a.sh
-p: select partition
